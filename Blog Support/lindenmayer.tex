\pgfdeclarelindenmayersystem{Dragon curve}{
	\rule{X -> X+YF+}
	\rule{Y -> -FX-Y}}
\pgfdeclarelindenmayersystem{quadric Koch island}{
	\rule{F -> F+F-F-FF+F+F-F}
}
\pgfdeclarelindenmayersystem{peano curve}{
	\rule{X -> XFYFX+F+YFXFY-F-XFYFX}
	\rule{Y -> YFXFY-F-XFYFX+F+YFXFY}
}
\pgfdeclarelindenmayersystem{sierpinski curve}{
	\rule{X -> XF-F+F-XF+F+XF-F+F-X}
}
\pgfdeclarelindenmayersystem{Hilbert curve}{
	\rule{X -> -YF+XFX+FY-}
	\rule{Y -> +XF-YFY-FX+}
}
\pgfdeclarelindenmayersystem{Sierpinski arrowhead}{
	\rule{G -> F+G+F}
	\rule{F -> G-F-G}
}
\pgfdeclarelindenmayersystem{Gosper curve}{
	\rule{XF -> XF+YF++YF-XF--XFXF-YF+}
	\rule{YF -> -XF+YFYF++YF+XF--XF-YF}
}
\pgfdeclarelindenmayersystem{mat}{
	\rule{F -> FF+F+F+F+FF}
}
\pgfdeclarelindenmayersystem{B}{
	\rule{F -> FF+F+F+F+F+F-F}
}
\pgfdeclarelindenmayersystem{plant}{
	\rule{X -> F-[[X]+x]+F[+FX]-X}
	\rule{F -> FF}
}
\pgfdeclarelindenmayersystem{weirdtree}{
	\rule{F -> FF+F++F+F}
}
\pgfdeclarelindenmayersystem{testrhombus}{
	\rule{F -> F+FF++FF+F+++F++F+++F}
}
\pgfdeclarelindenmayersystem{levyCurve}{
	\rule{F -> F+F+}
}
\pgfdeclarelindenmayersystem{leaf}{
	\rule{F -> F--F+F-}
}
\pgfdeclarelindenmayersystem{trefoil}{
	\rule{F -> FF-F}
}
\pgfdeclarelindenmayersystem{bush}{
	\rule{F -> FF[-F+F][+F-F]}
}
