\begin{exo}
We give the differential equation:
\[ \frac{dx}{dt}=6x-2x^2 .\]
\begin{enumerate}
\item How do we call this differential equation ?
\vspace*{1.5cm}
\item What are the critical points ? Use a phase diagram to determine wether each critical point is stable or unstable.
\vspace*{4cm}
\item Solve this differential equation with $x_0=1$.
\vspace*{8cm}
\end{enumerate}
\end{exo}



\begin{exo}
Solve the differential equation:
\[ y^{(3)}+y''-y'+15y=0,\]
using the fact that the function $x\mapsto e^{-3x}$ is solution of this differential equation.\\
Then find the unique solution satisfying the initial conditions:
\[ y(0)=0, \ y'(0)=1, \ y''(0)=3.\]
\end{exo}

\vspace*{10cm}

\begin{exo}
Solve the initial value problem:
\[ y^{(3)}-4y''+4y'=0, \quad y(0)=0, \ y'(0)=1, \ y''(0)=2.\]
\end{exo}
