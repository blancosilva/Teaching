\begin{exo}[Laplace Transform of $\sin kt$ and $\cos kt$]
\begin{enumerate} \item $a)$ Consider the function $f(t)=\sin kt$. Write the differential equation satisfied by $f$ (you will need to differentiate two times). What are the initial conditions ?\\
                        $b)$ Deduce the Laplace transform of $\sin kt$ (don't use the table of Laplace transform !!).
                  \item Use the Laplace transform of a derivate to deduce the Laplace transform of $\cos kt$.
\end{enumerate}
\end{exo}

\begin{exo}
Find the inverse Laplace transform of the following functions:
\[\begin{array}{l}
\ds{ a)\ F(s)=\frac{3s+5}{s^2+4}, \qquad b)\ G(s)=\frac{3}{s^2(s^2+9)} \quad \mbox{(without partial fractions)},}\\
\ds{ c)\ H(s)=\frac{2s+9}{s^2-6s+36}, \qquad d)\ K(s)=\frac{2s-1}{s^2-s-12} \quad \mbox{(with partial fractions)}}.
\end{array}\]
\end{exo}
