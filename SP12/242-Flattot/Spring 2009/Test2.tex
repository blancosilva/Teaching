\item Solve this differential equation with $x_0=2$.
\end{enumerate}
\end{exo}

\begin{exo}
Find a linear homogeneous constant-coefficient differential equation with, in each cases, the given general solution:
\begin{enumerate}[a)]
\item $y(x)=Ae^{-x}+Be^{2x}+Ce^x$,
\vspace*{3cm}
\item $y(x)=A+Bx+Cx^2+De^{17x}$,
\vspace*{3cm}
\item $y(x)= Ae^{3x}+B\cos(2x)+ C\sin(2x)+x(D\cos(2x)+E\sin(2x))$.
\end{enumerate}
\end{exo}

\begin{exo}
Solve the differential equation:
\[ y^{(3)}-5y''-16y'+80y=0,\]
using the fact that the function $x\mapsto e^{5x}$ is solution of this differential equation.\\
Then find the unique solution satisfying the initial conditions:
\[ y(0)=9/4, \ y'(0)=-3, \ y''(0)=18.\]
\end{exo}

\begin{exo}
We consider the initial value problem:
\[ y''+y'-2y=(20x+4)e^{3x}, \quad y(0)=4,\ y'(0)=1.\]
Using that a particular solution is given by $y_p(x)=(2x-1)e^{3x}$, solve this problem.
\end{exo}
