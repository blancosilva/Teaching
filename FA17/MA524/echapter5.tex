\section*{Exercises}

\begin{problem}[Basic]
Solve the following linear program by the simplex method:
\begin{equation*}
(LP): \begin{cases}
\displaystyle{\max_{(x,y,z) \in \field{R}^3} \big( 4x+y-z \big)} \\
x+3z \leq 6 \\
3x+y+3z \leq 9 \\
x \geq 0, y\geq 0, z\geq 0
\end{cases}
\end{equation*}
\end{problem}
